import { AngularFireUploadTask } from 'angularfire2/storage';
import { UploadTaskSnapshot } from 'angularfire2/storage/interfaces';
import { BehaviorSubject } from 'rxjs';
import { FileSnapshot } from './fire-uploader.model';
import { FireUploaderRef } from './fire-uploader-ref';
import { isImage, resizeImage } from './utils';

/**
 * FileItem class for each file added in the queue
 */
export class FileItem {

  /** AngularFireUploadTask ref */
  private _task: AngularFireUploadTask;

  /** FileItem state */
  snapshot: FileSnapshot;

  /** Stream that emits FileItem state */
  snapshot$ = new BehaviorSubject<FileSnapshot>({});

  /** Creates FileItem class for each file added to uploader queue */
  constructor(public file: File, private _uploader: FireUploaderRef) {

    this.updateSnapshot({
      name: file.name,
      type: file.type,
      active: false,
      extension: file.name.split('.').pop(),
      progress: {
        percentage: 0,
        bytesTransferred: 0,
        totalBytes: file.size
      }
    });

    // If file is type of image, create a thumbnail
    if (this._uploader.config.thumbs && isImage(file)) {

      resizeImage(this, this._uploader.config.thumbWidth, this._uploader.config.thumbHeight, this._uploader.config.thumbMethod, 1)
        .subscribe(
          (blob: Blob) => this.updateSnapshot({thumbnail: window.URL.createObjectURL(blob)}),
          (err: Error) => this._uploader.error$.next(err)
        );
    }
  }

  /** Assign AngularFireUploadTask to FileItem */
  assignTask(task: AngularFireUploadTask): Promise<any> {
    this._task = task;

    this._task.snapshotChanges()
      .subscribe((snapshot: UploadTaskSnapshot) => this.onSnapshotChanges(snapshot));

    return this._task
      .then((snapshot: any) => this.onTaskComplete(snapshot))
      .catch((err: Error) => {
        this.updateSnapshot({active: false});
        this._uploader.error$.next(err);
      });
  }

  /** Delete file after it is uploaded */
  delete() {
    this.snapshot.ref.delete()
      .then(() => {
        this._uploader.remove$.next(this);
        this.cancel();
      })
      .catch((err) => this._uploader.error$.next(err));
  }

  /** Pause upload task */
  pause() {
    if (this._task) {
      this._task.pause();
    }
  }

  /** Resume upload task */
  resume() {
    if (this._task) {
      this._task.resume();
    }
  }

  /** Cancel upload task */
  cancel() {
    if (this._task) {
      this._task.cancel();
    }
    this.reset();
  }

  /** Reset FileItem state */
  reset() {
    this.updateSnapshot({
      state: null,
      active: false,
      downloadURL: null,
      progress: {
        percentage: 0,
        bytesTransferred: 0,
        totalBytes: this.file.size
      }
    });
  }

  /** Update FileItem state */
  private updateSnapshot(snapshot: FileSnapshot) {
    this.snapshot = {...this.snapshot, ...snapshot};
    this.snapshot$.next(this.snapshot);
  }

  /** Update FileItem state when UploadTaskSnapshot changes */
  private onSnapshotChanges(snapshot: UploadTaskSnapshot) {
    this.updateSnapshot({
      active: snapshot.state === 'running',
      state: snapshot.state,
      progress: {
        percentage: (snapshot.bytesTransferred / snapshot.totalBytes) * 100,
        bytesTransferred: snapshot.bytesTransferred,
        totalBytes: snapshot.totalBytes
      }
    });
    this._uploader.updateRootState$.next(null);
  }

  /** Update FileItem state when UploadTaskSnapshot completes */
  private onTaskComplete(snapshot: any) {
    if (snapshot.downloadURL) {
      this.updateSnapshot({
        downloadURL: snapshot.downloadURL,
        ref: snapshot.ref,
        active: false,
        state: snapshot.state,
        progress: {
          percentage: (snapshot.bytesTransferred / snapshot.totalBytes) * 100,
          bytesTransferred: snapshot.bytesTransferred,
          totalBytes: snapshot.totalBytes
        }
      });
      this._uploader.success$.next(this);
    }
  }

}
